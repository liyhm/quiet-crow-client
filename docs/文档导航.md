```
quiet-crow-client/
├── 📄 README.md                    # 项目说明
├── 📄 文档导航.md                  # 本文件
├── 📄 快速开始.md                  # 快速入门
├── 📄 项目总结.md                  # 项目介绍
├── 📄 实现状态.md                  # 功能状态
├── 📄 后端集成完成.md              # 集成说明
├── 📄 集成总结.md                  # 集成详情
├── 📄 API集成指南.md               # API 使用
├── 📄 API测试指南.md               # API 测试
├── 📄 后端API文档.md               # 后端接口
├── 📄 环境配置说明.md              # 环境配置
├── 📄 部署检查清单.md              # 部署清单
├── 📄 .env.development             # 开发环境配置
├── 📄 .env.production              # 生产环境配置
├── 📄 .env.example                 # 配置模板
└── src/
    └── renderer/
        └── src/
            ├── shared/
            │   ├── config/
            │   │   └── api.ts          # API 配置
            │   ├── api/
            │   │   ├── request.ts      # HTTP 客户端
            │   │   ├── auth.ts         # 认证 API
            │   │   ├── user.ts         # 用户 API
            │   │   ├── session.ts      # 会话 API
            │   │   ├── encryption.ts   # 加密 API
            │   │   ├── contacts.ts     # 联系人 API
            │   │   └── websocket.ts    # WebSocket
            │   └── lib/
            │       └── cryptoBackend.ts # 加密工具
            └── entities/
                ├── user/
                │   └── model/
                │       └── useAuthStore.ts      # 认证状态
                ├── session/
                │   └── model/
                │       └── useSessionStore.ts   # 会话状态
                └── message/
                    └── model/
                        └── useMessageStore.ts   # 消息状态
```
