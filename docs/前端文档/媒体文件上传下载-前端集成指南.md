# 媒体文件上传下载 - 前端集成指南

## 📋 概述

本文档提供媒体文件上传下载功能的前端集成指南，包括完整的代码示例和最佳实践。

**功能特性**:
- 支持图片和视频上传下载
- 端到端加密（客户端加密，服务器存储加密文件）
- 文件大小限制：图片 5MB，视频 50MB
- 支持格式：JPEG, PNG, GIF, WebP, MP4, MOV, AVI, WebM
- 权限验证：仅会话成员可访问

## 📤 文件上传

### 1. 完整上传流程

```javascript
/**
 * 上传文件到服务器
 * @param {File} file - 原始文件对象
 * @param {string} sessionId - 会话ID
 * @param {string} sessionKey - 会话密钥（用于加密）
 * @returns {Promise<string>} 返回文件ID
 */
async function uploadMediaFile(file, sessionId, sessionKey) {
  try {
    // 1. 验证文件大小
    const maxSize = file.type.startsWith('image/') ? 5 * 1024 * 1024 : 50 * 1024 * 1024;
    if (file.size > maxSize) {
      throw new Error(`文件大小超过限制（${file.type.startsWith('image/') ? '5MB' : '50MB'}）`);
    }

    // 2. 验证文件类型
    const allowedTypes = {
      image: ['image/jpeg', 'image/png', 'image/gif', 'image/webp'],
      video: ['video/mp4', 'video/quicktime', 'video/x-msvideo', 'video/webm']
    };
    const isImage = allowedTypes.image.includes(file.type);
    const isVideo = allowedTypes.video.includes(file.type);
    
    if (!isImage && !isVideo) {
      throw new Error('不支持的文件类型');
    }

    // 3. 读取文件内容
    const fileBuffer = await file.arrayBuffer();

    // 4. 加密文件内容
    const encryptedFileBuffer = await encryptFile(fileBuffer, sessionKey);

    // 5. 加密文件名
    const encryptedFileName = await encryptText(file.name, sessionKey);
    const base64FileName = btoa(encryptedFileName);

    // 6. 准备表单数据
    const formData = new FormData();
    formData.append('file', new Blob([encryptedFileBuffer]), 'encrypted.bin');
    formData.append('metadata', JSON.stringify({
      sessionId: sessionId,
      fileName: base64FileName,
      fileSize: file.size,
      mimeType: file.type,
      messageType: isImage ? 'IMAGE' : 'VIDEO'
    }));

    // 7. 发送请求
    const response = await fetch(`${API_BASE_URL}/api/media/upload`, {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${token}`
      },
      body: formData
    });

    if (!response.ok) {
      const error = await response.json();
      throw new Error(error.message || '上传失败');
    }

    const result = await response.json();
    return result.data.fileId;

  } catch (error) {
    console.error('文件上传失败:', error);
    throw error;
  }
}
```

## 📥 文件下载

### 1. 完整下载流程

```javascript
/**
 * 下载并解密文件
 * @param {string} fileId - 文件ID
 * @param {string} sessionKey - 会话密钥（用于解密）
 * @returns {Promise<Blob>} 返回解密后的文件Blob
 */
async function downloadMediaFile(fileId, sessionKey) {
  try {
    // 1. 下载加密文件
    const response = await fetch(`${API_BASE_URL}/api/media/${fileId}`, {
      method: 'GET',
      headers: {
        'Authorization': `Bearer ${token}`
      }
    });

    if (!response.ok) {
      throw new Error('文件下载失败');
    }

    // 2. 获取文件元数据
    const encryptedFileName = response.headers.get('X-File-Name');
    const mimeType = response.headers.get('X-Mime-Type');
    const fileSize = response.headers.get('X-File-Size');
    const uploadTime = response.headers.get('X-Upload-Time');

    // 3. 读取加密数据
    const encryptedData = await response.arrayBuffer();

    // 4. 解密文件内容
    const decryptedData = await decryptFile(encryptedData, sessionKey);

    // 5. 解密文件名
    const base64Decoded = atob(encryptedFileName);
    const originalFileName = await decryptText(base64Decoded, sessionKey);

    // 6. 创建 Blob
    const blob = new Blob([decryptedData], { type: mimeType });

    // 返回文件信息
    return {
      blob,
      fileName: originalFileName,
      mimeType,
      fileSize: parseInt(fileSize),
      uploadTime
    };

  } catch (error) {
    console.error('文件下载失败:', error);
    throw error;
  }
}
```

async function downloadToLocal(fileId, sessionKey) {

    // 创建下载链接
    const url = URL.createObjectURL(blob);

    a.download = fileName;
    document.body.appendChild(a);
    a.click();
    document.body.removeChild(a);

    setTimeout(() => URL.revokeObjectURL(url), 100);

  } catch (error) {
    showError('文件下载失败');

  }
}

## 🔐 加密解密函数

### 使用 Web Crypto API

```javascript
/**
 * 加密文件内容
 * @param {ArrayBuffer} fileBuffer - 文件内容
 * @param {string} sessionKey - 会话密钥（Base64）
 * @returns {Promise<ArrayBuffer>} 加密后的内容
 */
async function encryptFile(fileBuffer, sessionKey) {
  // 1. 导入密钥
  const keyData = Uint8Array.from(atob(sessionKey), c => c.charCodeAt(0));
  const cryptoKey = await crypto.subtle.importKey(
    'raw',
    keyData,
    { name: 'AES-GCM' },
    false,
    ['encrypt']
  );

  // 2. 生成随机 IV
  const iv = crypto.getRandomValues(new Uint8Array(12));

  // 3. 加密
  const encrypted = await crypto.subtle.encrypt(
    { name: 'AES-GCM', iv },
    cryptoKey,
    fileBuffer
  );

  // 4. 合并 IV 和加密数据
  const result = new Uint8Array(iv.length + encrypted.byteLength);
  result.set(iv, 0);
  result.set(new Uint8Array(encrypted), iv.length);

  return result.buffer;
}

/**
 * 解密文件内容
 * @param {ArrayBuffer} encryptedBuffer - 加密的内容
 * @param {string} sessionKey - 会话密钥（Base64）
 * @returns {Promise<ArrayBuffer>} 解密后的内容
 */
async function decryptFile(encryptedBuffer, sessionKey) {
  // 1. 导入密钥
  const keyData = Uint8Array.from(atob(sessionKey), c => c.charCodeAt(0));
  const cryptoKey = await crypto.subtle.importKey(
    'raw',
    keyData,
    { name: 'AES-GCM' },
    false,
    ['decrypt']
  );

  // 2. 提取 IV 和加密数据
  const data = new Uint8Array(encryptedBuffer);
  const iv = data.slice(0, 12);
  const encrypted = data.slice(12);

  // 3. 解密
  const decrypted = await crypto.subtle.decrypt(
    { name: 'AES-GCM', iv },
    cryptoKey,
    encrypted
  );

  return decrypted;
}

/**
 * 加密文本
 * @param {string} text - 原始文本
 * @param {string} sessionKey - 会话密钥（Base64）
 * @returns {Promise<string>} 加密后的文本
 */
async function encryptText(text, sessionKey) {
  const encoder = new TextEncoder();
  const data = encoder.encode(text);
  const encrypted = await encryptFile(data.buffer, sessionKey);
  return btoa(String.fromCharCode(...new Uint8Array(encrypted)));
}

/**
 * 解密文本
 * @param {string} encryptedText - 加密的文本（Base64）
 * @param {string} sessionKey - 会话密钥（Base64）
 * @returns {Promise<string>} 解密后的文本
 */
async function decryptText(encryptedText, sessionKey) {
  const data = Uint8Array.from(atob(encryptedText), c => c.charCodeAt(0));
  const decrypted = await decryptFile(data.buffer, sessionKey);
  const decoder = new TextDecoder();
  return decoder.decode(decrypted);
}
```

/**
 * 发送媒体消息
 * @param {File} file - 文件对象
 * @param {string} sessionId - 会话ID

 */
async function sendMediaMessage(file, sessionId, sessionKey) {
  try {
    // 1. 上传文件
    const fileId = await uploadMediaFile(file, sessionId, sessionKey);

 * 处理接收到的媒体消息
 * @param {Object} message - 消息对象

async function handleMediaMessage(message) {
  try {
    // 1. 解密文件ID
    const fileId = await decryptText(message.content, sessionKey);

    // 2. 创建消息容器
    const messageElement = createMessageElement(message);
    const mediaContainer = messageElement.querySelector('.media-container');

      await displayImage(fileId, sessionKey, mediaContainer);

    mediaContainer.appendChild(downloadBtn);

  } catch (error) {
    console.error('媒体消息处理失败:', error);

async function uploadWithProgress(file, sessionId, sessionKey, onProgress) {
  // 使用 XMLHttpRequest 以支持进度回调
  return new Promise((resolve, reject) => {
    const xhr = new XMLHttpRequest();

    xhr.upload.addEventListener('progress', (e) => {
      if (e.lengthComputable) {
        const percent = (e.loaded / e.total) * 100;
        onProgress(percent);
      }
    });

    xhr.addEventListener('load', () => {
      if (xhr.status === 200) {
        const result = JSON.parse(xhr.responseText);
        resolve(result.data.fileId);
      } else {
        reject(new Error('上传失败'));
      }
    });

    xhr.addEventListener('error', () => reject(new Error('网络错误')));

    // 准备表单数据...

    xhr.setRequestHeader('Authorization', `Bearer ${token}`);

## 🆘 常见问题

### Q1: 上传失败，提示"无权限"
**原因**: 用户不是会话成员  
**解决**: 确保用户已加入该会话

### Q2: 下载的图片无法显示
**原因**: 解密失败或密钥错误  
**解决**: 检查会话密钥是否正确

### Q3: 视频播放卡顿
**原因**: 文件过大或网络慢  
**解决**: 建议压缩视频或使用流式传输

### Q4: 文件上传后找不到
**原因**: 文件ID未正确保存  
**解决**: 确保上传成功后保存返回的 fileId
