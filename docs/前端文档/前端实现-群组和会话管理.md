## 已实现功能

### 1. 删除会话

**文件**：`src/renderer/src/entities/session/model/useSessionStore.ts`

**功能**：
- 调用后端 API 删除会话
- 从本地列表中移除会话
- 如果是当前活动会话，清除选中状态

**代码**：
```typescript
const deleteSession = async (sessionId: string) => {
  try {
    await sessionApi.deleteSession(sessionId)
    
    if (activeSessionId.value === sessionId) {
      activeSessionId.value = null
    }
    
    sessions.value = sessions.value.filter((s) => s.id !== sessionId)
    
    return { success: true }
  } catch (error) {
    return { success: false, error: String(error) }
  }
}
```

### 2. 接收消息时自动恢复会话

**文件**：`src/renderer/src/shared/api/websocket.ts`

**功能**：
- 收到消息时检查会话是否存在
- 如果不存在，重新加载会话列表
- 后端会自动恢复 `is_deleted = false`

**代码**：
```typescript
private async handleIncomingMessage(message: any) {
  const sessionStore = useSessionStore()
  
  // 检查会话是否存在
  const sessionExists = sessionStore.sessions.find(s => s.id === message.sessionId)
  
  if (!sessionExists) {
    console.log('⚠️ 会话不存在，重新加载会话列表')
    await sessionStore.loadSessions()
  }
  
  // 处理消息...
}
```

import { ref } from 'vue'
import { useContactStore } from '@/entities/contact/model/useContactStore'
import { useSessionStore } from '@/entities/session/model/useSessionStore'

const emit = defineEmits(['close', 'created'])

const contactStore = useContactStore()
const sessionStore = useSessionStore()

const groupName = ref('')
const selectedMembers = ref<string[]>([])

const contacts = computed(() => contactStore.contacts)

const toggleMember = (userId: string) => {
  const index = selectedMembers.value.indexOf(userId)
  if (index > -1) {
    selectedMembers.value.splice(index, 1)
  } else {
    selectedMembers.value.push(userId)
  }
}

const handleCreate = async () => {
  const result = await sessionStore.createGroupSession(
    groupName.value,
    selectedMembers.value
  )
  
  if (result.success) {
    emit('created', result.sessionId)
    emit('close')
  } else {
    alert('创建群聊失败: ' + result.error)
  }
}
</script>

## API 端点配置

文件：`src/renderer/src/shared/config/api.ts`

```typescript
export const API_ENDPOINTS = {
  SESSION: {
    MY: '/api/session/my',
    PRIVATE: '/api/session/private',
    GROUP: '/api/session/group',
    MESSAGES: (sessionId: string) => `/api/session/${sessionId}/messages`,
    READ: (sessionId: string) => `/api/session/${sessionId}/read`,
    READ_SINGLE: (messageId: string) => `/api/session/message/${messageId}/read`,
    READ_BATCH: (sessionId: string) => `/api/session/${sessionId}/read/batch`,
    DELETE: (sessionId: string) => `/api/session/${sessionId}`,
    
    // 群聊相关
    LEAVE: (sessionId: string) => `/api/session/${sessionId}/leave`,
    DISBAND: (sessionId: string) => `/api/session/${sessionId}/disband`,
    MEMBERS: (sessionId: string) => `/api/session/${sessionId}/members`,
    ADD_MEMBERS: (sessionId: string) => `/api/session/${sessionId}/members`,
    REMOVE_MEMBER: (sessionId: string, userId: string) => 
      `/api/session/${sessionId}/members/${userId}`
  }
}
```
