## 🔐 加密实现细节

### 加密流程

```typescript
// 1. 加密文件
const encryptedData = await FileCryptoService.encryptFile(file, sessionKey)
// 返回: [12字节 IV][加密数据]

// 2. 加密文件名
const encryptedFileName = await FileCryptoService.encryptText(fileName, sessionKey)
// 返回: Base64([12字节 IV][加密数据])

// 3. 上传
await mediaApi.uploadMedia({
  file: new Blob([encryptedData]),
  metadata: {
    sessionId,
    fileName: encryptedFileName,  // Base64 编码
    fileSize: originalSize,
    mimeType: file.type,
    messageType: 'IMAGE' | 'VIDEO'
  }
})
```

### 解密流程

```typescript
// 1. 下载加密文件
const response = await mediaApi.downloadMedia(fileId)

// 2. 解密文件
const decryptedData = await FileCryptoService.decryptFile(
  response.data,  // 包含 IV 的 ArrayBuffer
  sessionKey
)

// 3. 解密文件名
const fileName = await FileCryptoService.decryptText(
  response.fileName,  // Base64 编码
  sessionKey
)

// 4. 创建 Blob
const blob = new Blob([decryptedData], { type: response.mimeType })
```

## 🔧 代码示例

### 上传文件

```typescript
// 在 MainPage.vue 中
const uploadFile = async (file: File) => {
  // 1. 获取会话密钥
  const sessionKey = await getSessionKey(sessionId)
  
  // 2. 加密文件
  const encryptedData = await FileCryptoService.encryptFile(file, sessionKey)
  
  // 3. 加密文件名
  const encryptedFileName = await FileCryptoService.encryptText(file.name, sessionKey)
  
  // 4. 上传
  const result = await mediaApi.uploadMedia({
    file: new Blob([encryptedData]),
    metadata: {
      sessionId,
      fileName: encryptedFileName,
      fileSize: file.size,
      mimeType: file.type,
      messageType: file.type.startsWith('image/') ? 'IMAGE' : 'VIDEO'
    }
  }, (progress) => {
    console.log('上传进度:', progress + '%')
  })
  
  // 5. 发送消息
  const messageContent = JSON.stringify({
    fileId: result.fileId,
    fileName: file.name,
    fileSize: file.size
  })
  
  websocketService.sendMessage(sessionId, {
    sessionId,
    content: encryptWithSession(sessionId, messageContent),
    encrypted: true,
    messageType: file.type.startsWith('image/') ? '2' : '3'
  })
}
```

### 下载文件

```typescript
// 下载并显示图片
const displayImage = async (fileId: string) => {
  // 1. 获取会话密钥
  const sessionKey = await getSessionKey(sessionId)
  
  // 2. 下载文件
  const response = await mediaApi.downloadMedia(fileId)
  
  // 3. 解密文件
  const decryptedData = await FileCryptoService.decryptFile(
    response.data,
    sessionKey
  )
  
  // 4. 创建 Blob URL
  const blob = new Blob([decryptedData], { type: response.mimeType })
  const url = URL.createObjectURL(blob)
  
  // 5. 显示图片
  imgElement.src = url
}
```

```typescript
// 及时释放 Blob URL
const cleanup = () => {
  if (blobUrl) {
    URL.revokeObjectURL(blobUrl)
  }

// 组件卸载时清理
onUnmounted(() => {
  cleanup()
})
```

### 2. 错误处理
```typescript
try {
  await uploadFile(file)
} catch (error) {
  if (error.message.includes('网络')) {
    showAlert('网络错误，请检查连接', 'error')
  } else if (error.message.includes('大小')) {
    showAlert('文件过大', 'error')
  } else {
    showAlert('上传失败: ' + error.message, 'error')
  }
}
```
