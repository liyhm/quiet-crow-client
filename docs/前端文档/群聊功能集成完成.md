## 💡 使用示例

### 创建群聊
```vue
<CreateGroupModal
  :show="showCreateGroupModal"
  :contacts="contactList"
  @close="showCreateGroupModal = false"
  @create="handleCreateGroup"
/>

<script>
const handleCreateGroup = async (data: { name: string; memberIds: string[] }) => {
  const response = await sessionApi.createGroup(data.name, data.memberIds)
  if (response.code === 200) {
    showCreateGroupModal.value = false
    // 刷新会话列表
  }
}
</script>
```

### 显示群成员
```vue
<GroupMembersModal
  :show="showGroupMembersModal"
  :group-name="activeSession.name"
  :members="groupMembers"
  @close="showGroupMembersModal = false"
/>
```

### 群聊消息气泡
```vue
<MessageBubble
  :content="message.content"
  :timestamp="message.timestamp"
  :is-mine="message.senderId === currentUserId"
  :sender-name="getSenderName(message.senderId)"
  :is-group="activeSession.type === 'GROUP'"
/>
```
